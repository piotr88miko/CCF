216 template <typename In, typename Out>
217 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
218 {
219 if constexpr (!std::is_same_v<In, void>)
220 {
221 params_schema = ds::json::build_schema<In>();
222
223 schema_builders.push_back(
224 [](nlohmann::json& document, const Endpoint& endpoint) {
225 const auto http_verb = endpoint.dispatch.verb.get_http_method();
226 if (!http_verb.has_value())
227 {
228 // Non-HTTP endpoints are not documented
229 return;
230 }
231
232 ds::openapi::add_request_body_schema<In>(
233 document, endpoint.full_uri_path, http_verb.value());
234 });
235 }
236 else
237 {
238 params_schema = nullptr;
239 }
240
241 if constexpr (!std::is_same_v<Out, void>)
242 {
243 success_status = status.value_or(HTTP_STATUS_OK);
244
245 result_schema = ds::json::build_schema<Out>();
246
247 schema_builders.push_back(
248 [](nlohmann::json& document, const Endpoint& endpoint) {
249 const auto http_verb = endpoint.dispatch.verb.get_http_method();
250 if (!http_verb.has_value())
251 {
252 return;
253 }
254
255 ds::openapi::add_response_schema<Out>(
256 document,
257 endpoint.full_uri_path,
258 http_verb.value(),
259 endpoint.success_status);
260 });
261 }
262 else
263 {
264 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
265 result_schema = nullptr;
266 }
267
268 return *this;
269 }

302 template <typename T>
303 Endpoint& add_query_parameter(
304 const std::string& param_name,
305 QueryParamPresence presence = RequiredParameter)
306 {
307 schema_builders.push_back(
308 [param_name,
309 presence](nlohmann::json& document, const Endpoint& endpoint) {
310 const auto http_verb = endpoint.dispatch.verb.get_http_method();
311 if (!http_verb.has_value())
312 {
313 // Non-HTTP endpoints are not documented
314 return;
315 }
316
317 const auto schema_name = ds::json::schema_name<T>();
318 const auto query_schema = ds::json::build_schema<T>();
319
320 auto parameter = nlohmann::json::object();
321 parameter["name"] = param_name;
322 parameter["in"] = "query";
323 parameter["required"] = presence == RequiredParameter;
324 parameter["schema"] = ds::openapi::add_schema_to_components(
325 document, schema_name, query_schema);
326 ds::openapi::add_request_parameter_schema(
327 document, endpoint.full_uri_path, http_verb.value(), parameter);
328 });
329
330 return *this;
331 }
