51 void reply(http::Response& r, const std::string& body)

53 if (!socket.is_null())
54 {
55 r.set_body(body);

61 virtual void on_read(size_t len, uint8_t*& incoming, sockaddr sa) override
62 {
63 try
64 {
65 parser.execute(incoming, len);
66 while (!sp.received.empty())
67 {
68 auto req = sp.received.front();
69 sp.received.pop();
70
71 // We serve only
72 // http://<YOUR_DOMAIN>/.well-known/acme-challenge/<TOKEN>
73 if (req.url.find("/.well-known/acme-challenge/") != 0)
74 {
75 std::string content((char*)incoming, len);
76 LOG_INFO_FMT(
77 "ACME: invalid request from {} for url={} with following "
78 "body:\n{}",
79 socket->get_peer_name(),
80 req.url,
81 content);
82 http::Response r(HTTP_STATUS_NOT_FOUND);
83 reply(r, "Not found");
84 }
85 else
86 {
87 auto token = req.url.substr(req.url.rfind('/') + 1);
88
89 if (token.empty())
90 {
91 throw std::runtime_error(fmt::format(
92 "Missing ACME token in {} (requested by {})",
93 req.url,
94 socket->get_peer_name()));
95 }
96
97 {
98 std::unique_lock<std::mutex> guard(lock);
99
100 std::string response;
101
102 auto tit = prepared_responses.find(token);
103 if (tit == prepared_responses.end())
104 {
105 auto prit = prepared_responses.find("");
106 if (prit != prepared_responses.end())
107 {
108 LOG_TRACE_FMT("ACME: using blanket response");
109 response = token + "." + prit->second;
110 }
111 else
112 {
113 LOG_DEBUG_FMT(
114 "ACME: challenge response for token '{}' not found "
115 "(requested "
116 "by {})",
117 token,
118 socket->get_peer_name());
119 http::Response r(HTTP_STATUS_NOT_FOUND);
120 reply(
121 r, fmt::format("No response for token '{}' found", token));
122 }
123 }
124 else
125 {
126 response = tit->second;
127 }
128
129 auto rbody = fmt::format("{}.{}", token, response);
130 http::Response r(HTTP_STATUS_OK);
131 r.set_header("Content-Type", "application/octet-stream");
132 reply(r, rbody);
133 LOG_DEBUG_FMT(
134 "ACME: challenge response for token '{}' provided to {}",
135 token,
136 socket->get_peer_name());
137 }
138 }
139 }
140 }
141 catch (const std::exception& ex)
142 {
143 LOG_INFO_FMT("HTTP handling failed with: {}", ex.what());
144 http::Response r(HTTP_STATUS_BAD_REQUEST);
145 reply(r, "Bad request");
146 }
147 }
148
149 protected:

191class ACMEChallengeServer : public ACMEConnectionTracker
192{
193public:
194 ACMEChallengeServer(
195 const std::string& interface,
196 messaging::Dispatcher<ringbuffer::Message>& disp,
197 ringbuffer::AbstractWriterFactory& writer_factory) :
198 listener(nullptr),
199 to_enclave(writer_factory.create_writer_to_inside())
200 {
201 auto iface = cli::validate_address(interface, "80");
202 host = iface.first;
203 port = iface.second;
204
205 DISPATCHER_SET_MESSAGE_HANDLER(
206 disp,
207 ACMEMessage::acme_challenge_response,
208 [this](const uint8_t* data, size_t size) {
209 try
210 {
211 auto [response] =
212 ringbuffer::read_message<ACMEMessage::acme_challenge_response>(
213 data, size);
214
215 auto dotidx = response.find(".");
216
217 std::string token, token_response;
218 if (dotidx != std::string::npos)
219 {
220 token = response.substr(0, dotidx);
221 token_response = response.substr(dotidx + 1);
222 }
223 else
224 {
225 token = "";
226 token_response = response;
227 }
228
229 LOG_TRACE_FMT(
230 "ACME: challenge server received response for token '{}' ({})",
231 token,
232 token_response);
233
234 {
235 std::unique_lock<std::mutex> guard(lock);
236 prepared_responses.emplace(token, token_response);
237 }
238
239 if (listener.is_null())
240 {
241 listener = asynchost::TCP();
242 listener->set_behaviour(std::make_unique<ACMEServerBehaviour>(
243 *this, lock, prepared_responses, to_enclave));
244 listener->listen(host, port);
245 }
246
247 RINGBUFFER_WRITE_MESSAGE(
248 ACMEMessage::acme_challenge_response_ack, to_enclave, token);
249 }
250 catch (const std::exception& ex)
251 {
252 LOG_FAIL_FMT(
253 "ACME: acme_challenge_response message handler failed: {}",
254 ex.what());
255 }
256 });
257
258 DISPATCHER_SET_MESSAGE_HANDLER(
259 disp,
260 ACMEMessage::acme_challenge_server_stop,
261 [this](const uint8_t* data, size_t size) {
262 listener = nullptr;
263 prepared_responses.clear();
264 LOG_DEBUG_FMT("ACME: challenge server stopped");
265 });
266 }
267
268 virtual ~ACMEChallengeServer() = default;
269
270 virtual void add(asynchost::TCP& peer) override
271 {
272 sockets.emplace(sockets.size(), peer);
273 }
274
275protected:
276 std::string host, port;
277 asynchost::TCP listener;
278 std::unordered_map<size_t, asynchost::TCP> sockets;
279 std::mutex lock;
280 std::map<std::string, std::string> prepared_responses;
281 ringbuffer::WriterPtr to_enclave;
