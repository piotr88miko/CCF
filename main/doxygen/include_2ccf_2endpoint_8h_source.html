199 template <typename In, typename Out>
200 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
201 {
202 if constexpr (!std::is_same_v<In, void>)
203 {
204 params_schema = ds::json::build_schema<In>();
205
206 schema_builders.push_back(
207 [](nlohmann::json& document, const Endpoint& endpoint) {
208 const auto http_verb = endpoint.dispatch.verb.get_http_method();
209 if (!http_verb.has_value())
210 {
211 // Non-HTTP endpoints are not documented
212 return;
213 }
214
215 ds::openapi::add_request_body_schema<In>(
216 document, endpoint.full_uri_path, http_verb.value());
217 });
218 }
219 else
220 {
221 params_schema = nullptr;
222 }
223
224 if constexpr (!std::is_same_v<Out, void>)
225 {
226 success_status = status.value_or(HTTP_STATUS_OK);
227
228 result_schema = ds::json::build_schema<Out>();
229
230 schema_builders.push_back(
231 [](nlohmann::json& document, const Endpoint& endpoint) {
232 const auto http_verb = endpoint.dispatch.verb.get_http_method();
233 if (!http_verb.has_value())
234 {
235 return;
236 }
237
238 ds::openapi::add_response_schema<Out>(
239 document,
240 endpoint.full_uri_path,
241 http_verb.value(),
242 endpoint.success_status);
243 });
244 }
245 else
246 {
247 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
248 result_schema = nullptr;
249 }
250
251 return *this;
252 }

285 template <typename T>
286 Endpoint& add_query_parameter(
287 const std::string& param_name,
288 QueryParamPresence presence = RequiredParameter)
289 {
290 schema_builders.push_back(
291 [param_name,
292 presence](nlohmann::json& document, const Endpoint& endpoint) {
293 const auto http_verb = endpoint.dispatch.verb.get_http_method();
294 if (!http_verb.has_value())
295 {
296 // Non-HTTP endpoints are not documented
297 return;
298 }
299
300 const auto schema_name = ds::json::schema_name<T>();
301 const auto query_schema = ds::json::build_schema<T>();
302
303 auto parameter = nlohmann::json::object();
304 parameter["name"] = param_name;
305 parameter["in"] = "query";
306 parameter["required"] = presence == RequiredParameter;
307 parameter["schema"] = ds::openapi::add_schema_to_components(
308 document, schema_name, query_schema);
309 ds::openapi::add_request_parameter_schema(
310 document, endpoint.full_uri_path, http_verb.value(), parameter);
311 });
312
313 return *this;
314 }
